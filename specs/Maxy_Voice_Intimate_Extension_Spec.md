/**

function stripAIHedging(text: string): string {
  // Remove patterns like "I'd be happy to..." "If you'd like..." 
  // "I should note that..." that leak through in explicit contexts
  const hedgingPatterns = [
    /^(I'd be happy to|I should mention|It's worth noting|I want to be clear|Just to clarify)[^.]*\.\s*/i,

    /^(Of course|Certainly|Absolutely)[,!]\s*/i,
  ];
  
  let cleaned = text;
  for (const pattern of hedgingPatterns) {
    cleaned = cleaned.replace(pattern, '');
  }
  return cleaned.trim();
}
